<Card title="POST /udm/v1/case/{{case_no}}/status" horizontal>
</Card>

Hosts can use this endpoint to retrieve the current status of the case.

    </ResponseField>

     value
    </ResponseField>

Path Parameters
---
<ResponseField name="case_no" type="String" required>
      Unique identifier for the case
</ResponseField>

Response
--- 
<ResponseField name="Case Status" type="CaseStatus Object">
  <Expandable title="properties">

    <ResponseField name="case_no" type="string" required>
     Unique identifier for the case
    </ResponseField>

    <ResponseField name="dispute_type" type="enum" required>
    Type of the dispute. Allowed values 1-Chargeback, 2-Pre-compliance, 3-Good faith, 4-Pre-Arbitration, 5-Arbitration, 6-Compliance
    </ResponseField>

     <ResponseField name="dispute_amount" type="string" required>
      Amount associated with the dispute
    </ResponseField>

    <ResponseField name="transaction_amount" type="string" required>
    Amount associated with the transaction
    </ResponseField>

     <ResponseField name="created_date" type="string" required>
      Dispute created date within UDM
    </ResponseField>

    <ResponseField name="status" type="string" required>
    Status of the dispute within UDM
    </ResponseField>

     <ResponseField name="dispute_category" type="string" required>
     Category for the dispute
    </ResponseField>

    <ResponseField name="sub_status" type="string" required>
    Sub-Status of the dispute within UDM
    </ResponseField>

     <ResponseField name="transaction_id" type="string" required>
    Transaction Identifier
    </ResponseField>

<ResponseField name="network_data" type="NetworkData Object">
  <Expandable title="properties">

    <ResponseField name="schemeDeadline" type="string" required>
    Deadline for responding to the case
    </ResponseField>

    <ResponseField name="network_submitted_date" type="string" required>
    Date when the case was submitted to the network
    </ResponseField>

    <ResponseField name="represented_amount" type="string" required>
    Amount represented by the acquirer
    </ResponseField>

    <ResponseField name="represented_date" type="string" required>
    Date when the case was represented by the network
    </ResponseField>

    <ResponseField name="reject_reason" type="string" required>
    Reason for rejection by the acquirer
    </ResponseField>

  </Expandable>
</ResponseField>

<ResponseField name="merchant_data" type="MerchanData Object">
  <Expandable title="properties">

  <ResponseField name="merchant_name" type="string" required>
   Name of the merchant
    </ResponseField>

    <ResponseField name="mcc" type="string" required>
    Merchant category code
    </ResponseField>

  </Expandable>
</ResponseField>

</Expandable>
</ResponseField>
